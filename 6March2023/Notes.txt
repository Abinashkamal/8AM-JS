use break with switch. 

continue:

direct array functions.

var arr= [1,2,4,5,5,6,6,6,4,2,4,5,4,3,4,5,5,674,245,6,7,8];


<html>

        var newArr = []; // 1 2 4 5 6

        for (var i = 0; i < arr.length; i++) {
            if (newArr.includes(arr[i])) {
                continue;
            }
            newArr.push(arr[i]);
        }
        arr = newArr;
